  git add *.html
  git add . (adicionar tudo)

  git commit...
  Abre o editor "vim"
  Apertar "i" para modo "insert"
  Escrever comentário de commit
  Apertar "esc"
  Digitar ":wq"
  Apertar "enter"

  git commit -m 'Alteração app.js"
